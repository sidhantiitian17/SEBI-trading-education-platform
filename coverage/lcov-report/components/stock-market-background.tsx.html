'use client';
 
import React, { useState, useEffect, useRef } from 'react';
 
interface StockData {
  symbol: string;
  price: number;
  change: number;
  changePercent: number;
}
 
interface ChartData {
  points: number[];
  color: string;
}
 
const StockMarketBackground: React.FC = () => {
  const [mousePosition, setMousePosition] = useState({ x: 0, y: 0 });
  const [marketData, setMarketData] = useState<StockData[]>([
    { symbol: 'NIFTY', price: 24500, change: 125, changePercent: 0.51 },
    { symbol: 'SENSEX', price: 79500, change: -200, changePercent: -0.25 },
    { symbol: 'RELIANCE', price: 2450, change: 61.25, changePercent: 2.56 },
    { symbol: 'TCS', price: 3200, change: -38.4, changePercent: -1.18 },
    { symbol: 'HDFC', price: 1650, change: 25, changePercent: 1.54 },
    { symbol: 'INFY', price: 1450, change: -15, changePercent: -1.02 },
    { symbol: 'ICICIBANK', price: 1100, change: 18, changePercent: 1.66 },
    { symbol: 'BHARTIARTL', price: 1400, change: 35, changePercent: 2.56 },
  ]);
 
  const chartData: ChartData[] = [
    { points: [0.2, 0.4, 0.3, 0.6, 0.5, 0.8, 0.7, 0.9], color: '#00d4ff' },
    { points: [0.1, 0.3, 0.2, 0.5, 0.4, 0.7, 0.6, 0.8], color: '#00ffff' },
    { points: [0.3, 0.5, 0.4, 0.7, 0.6, 0.9, 0.8, 0.95], color: '#ff6b9d' },
  ];
 
  const animationRef = useRef<number>();
  const debounceRef = useRef<NodeJS.Timeout>();
  const resizeDebounceRef = useRef<NodeJS.Timeout>();
  const canvasRef = useRef<HTMLCanvasElement>(null);
 
  // Mouse tracking
  useEffect(() => {
    const handleMouseMove = (e: MouseEvent) => {
      if (debounceRef.current) clearTimeout(debounceRef.current);
      debounceRef.current = setTimeout(() => {
        setMousePosition({ x: e.clientX, y: e.clientY });
      }, 50); // Debounce by 50ms
    };
 
    window.addEventListener('mousemove', handleMouseMove);
    return () => {
      window.removeEventListener('mousemove', handleMouseMove);
      if (debounceRef.current) clearTimeout(debounceRef.current);
    };
  }, []);
 
  // Update market data periodically
  useEffect(() => {
    const interval = setInterval(() => {
      setMarketData(prevData =>
        prevData.map(stock => {
          const oldPrice = stock.price;
          const priceDelta = (Math.random() - 0.5) * 10;
          const newPrice = oldPrice + priceDelta;
          const newChange = priceDelta;
          const newChangePercent = oldPrice !== 0 ? (priceDelta / oldPrice) * 100 : 0;
          return {
            ...stock,
            price: newPrice,
            change: newChange,
            changePercent: newChangePercent,
          };
        })
      );
    }, 3000);
 
    return () => clearInterval(interval);
  }, []);
 
  // Particle system
  const particles = useRef<Array<{
    x: number;
    y: number;
    vx: number;
    vy: number;
    life: number;
    maxLife: number;
  }>>([]);
 
  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
 
    const ctx = canvas.getContext('2d');
    if (!ctx) return;
 
    const resizeCanvas = () => {
      if (resizeDebounceRef.current) clearTimeout(resizeDebounceRef.current);
      resizeDebounceRef.current = setTimeout(() => {
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;
        initParticles(); // Reinitialize particles on resize
      }, 200); // Debounce by 200ms
    };
 
    const initParticles = () => {
      const particleCount = window.innerWidth < 768 ? 20 : 50;
      particles.current = Array.from({ length: particleCount }, () => ({
        x: Math.random() * canvas.width,
        y: Math.random() * canvas.height,
        vx: (Math.random() - 0.5) * 2,
        vy: (Math.random() - 0.5) * 2,
        life: Math.random() * 100,
        maxLife: 100,
      }));
    };
 
    resizeCanvas();
    window.addEventListener('resize', resizeCanvas);
    initParticles();
 
    const animateParticles = () => {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
 
      particles.current.forEach(particle => {
        particle.x += particle.vx;
        particle.y += particle.vy;
        particle.life--;
 
        if (particle.life <= 0) {
          particle.x = Math.random() * canvas.width;
          particle.y = Math.random() * canvas.height;
          particle.vx = (Math.random() - 0.5) * 2;
          particle.vy = (Math.random() - 0.5) * 2;
          particle.life = particle.maxLife;
        }
 
        // Wrap around edges
        if (particle.x > canvas.width) particle.x = 0;
        if (particle.x < 0) particle.x = canvas.width;
        if (particle.y > canvas.height) particle.y = 0;
        if (particle.y < 0) particle.y = canvas.height;
 
        ctx.beginPath();
        ctx.arc(particle.x, particle.y, 1, 0, Math.PI * 2);
        ctx.fillStyle = 'rgba(0, 212, 255, 0.6)';
        ctx.fill();
      });
 
      animationRef.current = requestAnimationFrame(animateParticles);
    };
 
    animateParticles();
 
    return () => {
      window.removeEventListener('resize', resizeCanvas);
      if (animationRef.current) {
        cancelAnimationFrame(animationRef.current);
      }
    };
  }, []);
 
  return (
    <div className="fixed inset-0 overflow-hidden pointer-events-none z-0">
      {/* Background Gradient */}
      <div className="absolute inset-0 bg-gradient-to-br from-[#1a1f3a] via-[#0f1425] to-[#1a1f3a]" />
 
      {/* Grid Overlay (Optimized with CSS) */}
      <div className="absolute inset-0 opacity-20" style={{
        backgroundImage: 'repeating-linear-gradient(0deg, transparent 0%, transparent 12.5%, rgba(255,255,255,0.1) 12.5%, rgba(255,255,255,0.1) calc(12.5% + 1px)), repeating-linear-gradient(90deg, transparent 0%, transparent 8.333%, rgba(255,255,255,0.1) 8.333%, rgba(255,255,255,0.1) calc(8.333% + 1px))',
        animation: 'pulse 2s infinite alternate',
      }} />
      <style jsx global>{`
        @keyframes pulse {
          0% { opacity: 0.2; }
          100% { opacity: 0.4; }
        }
      `}</style>
 
      {/* SEBI Logo */}
      <div className="absolute top-4 md:top-8 left-1/2 transform -translate-x-1/2 z-10">
        <div className="text-white text-2xl md:text-4xl font-bold tracking-wider opacity-80">
          SEBI
        </div>
      </div>
 
      {/* Price Ticker */}
      <div className="absolute top-16 md:top-20 left-0 right-0 overflow-hidden">
        <div className="flex animate-scroll-left whitespace-nowrap">
          {[...marketData, ...marketData].map((stock, index) => (
            <div
              key={`${stock.symbol}-${index}`}
              className="flex items-center space-x-2 md:space-x-4 mx-4 md:mx-8 text-white text-sm md:text-base"
            >
              <span className="font-semibold text-cyan-400 text-xs md:text-sm">{stock.symbol}</span>
              <span className="text-sm md:text-lg">₹{stock.price.toFixed(2)}</span>
              <span
                className={`text-xs md:text-sm ${
                  stock.change >= 0 ? 'text-green-400' : 'text-red-400'
                }`}
              >
                {stock.change >= 0 ? '+' : ''}{stock.change.toFixed(2)} ({stock.changePercent >= 0 ? '+' : ''}{stock.changePercent.toFixed(2)}%)
              </span>
            </div>
          ))}
        </div>
      </div>
 
      {/* Floating Charts */}
      <div className="absolute inset-0">
        {chartData.map((chart, index) => (
          <div
            key={index}
            className="absolute w-48 md:w-64 h-24 md:h-32 opacity-60 hover:opacity-100 transition-opacity duration-300 animate-float hidden md:block"
            style={{
              left: `${20 + index * 25}%`,
              top: `${30 + index * 20}%`,
              animationDelay: `${index * 0.5}s`,
            }}
          >
            <svg className="w-full h-full" viewBox="0 0 256 128">
              <path
                d={`M 0 ${128 - chart.points[0] * 128} ${chart.points
                  .map((point, i) => `L ${i * 32} ${128 - point * 128}`)
                  .join(' ')}`}
                stroke={chart.color}
                strokeWidth="2"
                fill="none"
                className="drop-shadow-lg animate-draw-line"
                style={{
                  filter: `drop-shadow(0 0 8px ${chart.color})`,
                }}
              />
            </svg>
          </div>
        ))}
      </div>
 
      <canvas ref={canvasRef} className="absolute inset-0 pointer-events-none" />
 
      {/* Mouse Follower */}
      <div
        className="absolute w-4 h-4 bg-cyan-400 rounded-full opacity-50 pointer-events-none"
        style={{
          left: `${mousePosition.x - 8}px`,
          top: `${mousePosition.y - 8}px`,
          transition: 'all 0.1s ease-out',
          boxShadow: '0 0 20px #00d4ff',
        }}
      />
 
      {/* Regulatory Disclaimer */}
      <div className="absolute bottom-2 md:bottom-4 left-2 md:left-4 right-2 md:right-4 text-white/60 text-xs md:text-sm text-center leading-tight">
        <div className="hidden md:block">
          SEBI Registered Investment Adviser | Past performance is not indicative of future results |
          Investment involves risk | Consult a financial advisor
        </div>
        <div className="md:hidden">
          SEBI Registered | Investment involves risk
        </div>
      </div>
    </div>
  );
};
 
export default StockMarketBackground;
